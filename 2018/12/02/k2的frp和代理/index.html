frp的两个设置

frps:这个就是server服务器的设置。

frpc:客户端设置

[abctcp] #设置名称，在多个用户时可用。
privilege_mode = true  #选中它时，在frps里不显示端口号。
remote_port = 7000    #对应的远程端口。也就是服务器端口，一般要改。
type = tcp    #模式设置。tcp,http,https...
#local_ip = 192.168.123.1     #一般情况可用，http_proxy时不用选用。 
#http_user = abc   这个选项是http模式下的自己设置用户名
#http_pwd = abc   这个选项是http模式下的自己设置的密码
#local_port = 22  映射的本地端口，也就是把这个口映射到7000的口。
use_gzip = true   #这是个压缩
subdomain = ujn   #很重要，自定义二级域名。
plugin = http_proxy     #本次实现的功能，转发内网的网络
plugin_http_user = abc   #转发时，加个账号密码，安全。
plugin_http_passwd = abc
